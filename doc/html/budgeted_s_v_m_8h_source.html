283  BUDGET_SIZE = 50; // maximum number of weights per class in AMM, OR SVM budget size in BSGD, OR number of landmark points in LLSVM

285  KERNEL = 0; // kernel to use in kernel-based algorithms, 0 - Gaussian, 1 - exponential, 2 - polynomial, 3 - linear, 4 - sigmoid, 5 - user-defined
286  KERNEL_GAMMA_PARAM = 0.0; // sigma scale parameter in Gaussian kernel (default value of 0.0 indicates that the Gaussian kernel width for RBF kernel is set to 1/dimensionality)
287  KERNEL_DEGREE_PARAM = 2.0; // degree parameter in polynomial kernel, OR slope parameter in sigmoid kernel
288  KERNEL_COEF_PARAM = 1.0; // coefficient (intercept) in polynomial and sigmoid kernels

294 
295  VERBOSE = 0; // verbose output

297  RANDOMIZE = 1; // randomize (i.e., shuffle) the training data, 1 - randimization on, 0 - randomization off
298  CHUNK_SIZE = 50000; // size of chunk of file loaded in budgetedData (when a file is too budget to fit in memory)
299  CHUNK_WEIGHT = 1000; // size of chunk of budgetedVector weight (since vector is split into many small parts)
300  VERY_SPARSE_DATA = 99; // for sparse data, we can speed up computations by directly computing kernels from budgetedData, 99 for default, 0 and 1 when set by user;
301  // 0 when a user wants all kernel computations done between budgetedVectors, 1 for computations between budgetedVectors and vectors

304  // is VERY_SPARSE_DATA = 1 if data sparsity is less than 5%, otherwise 0

314  void updateVerySparseDataParameter(double dataSparsity)
315  {
316  // if the parameter is already set then just return and change nothing; it can be that it was set by a user
317  // or it was already set when the earlier data chunks were loaded
318  if ((VERY_SPARSE_DATA == 0) || (VERY_SPARSE_DATA == 1))
