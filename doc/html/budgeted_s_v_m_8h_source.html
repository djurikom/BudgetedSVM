650  budgetedVector(unsigned int dim, unsigned int chnkWght)
651  {
652  dimension = dim;
653  arrayLength = 0;
654  if (chnkWght > 0)

703  virtual void createVectorUsingDataPoint(budgetedData* inputData, unsigned int t, parameters* param)
704  {
705  unsigned int ibegin = inputData->ai[t];
706  unsigned int iend = (t == (unsigned int) (inputData->ai.size() - 1)) ? (unsigned int) (inputData->aj.size()) : inputData->ai[t + 1];

708  this->clear();
