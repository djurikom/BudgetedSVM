24 void libsvmwrite(const char *filename, const mxArray *label_vec, const mxArray *instance_mat)
25 {
26  FILE *fp = fopen(filename,"w");
27  int i, k, low, high, l;
28  mwIndex *ir, *jc;
29  int label_vector_row_num;
30  double *samples, *labels;
31  mxArray *instance_mat_col; // instance sparse matrix in column format
32 
33  if(fp ==NULL)
34  {
35  mexPrintf("can't open output file %s\n",filename);
36  return;
37  }
38 
39  // transpose instance matrix
40  {
41  mxArray *prhs[1], *plhs[1];
42  prhs[0] = mxDuplicateArray(instance_mat);
43  if(mexCallMATLAB(1, plhs, 1, prhs, "transpose"))
44  {
45  mexPrintf("Error: cannot transpose instance matrix\n");
46  return;
47  }
48  instance_mat_col = plhs[0];
49  mxDestroyArray(prhs[0]);
50  }
51 
52  // the number of instance
53  l = (int) mxGetN(instance_mat_col);
54  label_vector_row_num = (int) mxGetM(label_vec);
55 
56  if(label_vector_row_num!=l)
57  {
58  mexPrintf("Length of label vector does not match # of instances.\n");
59  return;
60  }
61 
62  // each column is one instance
63  labels = mxGetPr(label_vec);
64  samples = mxGetPr(instance_mat_col);
65  ir = mxGetIr(instance_mat_col);
66  jc = mxGetJc(instance_mat_col);
67 
68  for(i=0;i<l;i++)
69  {
70  fprintf(fp,"%g", labels[i]);
71 
72  low = (int) jc[i], high = (int) jc[i+1];
73  for(k=low;k<high;k++)
74  fprintf(fp," %ld:%g", ir[k]+1, samples[k]);
75 
76  fprintf(fp,"\n");
77  }
78 
79  fclose(fp);
80  return;
81 }
