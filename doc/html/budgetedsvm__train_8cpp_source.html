38 void mexFunction(int nlhs, mxArray *plhs[], int nrhs, const mxArray *prhs[])
39 {
40  parameters param;
41  char trainFileName[1024];
42  char modelFileName[1024];
43  char inputParamString[1024];
44 
45  if (nrhs == 0)
46  {
47  printUsageMatlab(true, &param);
48  fakeAnswer(plhs);
49  return;
50  }
51 
52  // Check the number of inputs
53  if ((nrhs != 2) && (nrhs != 3))
54  mexErrMsgTxt("Error, wrong number of input parameters!\nRun 'budgetedsvm_train()' for help.");
55 
56  // init random number generator
57  srand((unsigned)time(NULL));
58 
59  // here we first check what kind of inputs are we working with, .txt files or Matlab variables; we will check that
60  // by checking the first input parameter, if it is text file then we conclude that we are working with text inputs
61  if ((!mxIsDouble(prhs[0])) && (!mxIsSparse(prhs[0])))
62  {
63  budgetedModel *model;
64  budgetedData *trainData;
65 
66  // input 1 - training data, loaded in the switch below
67  mxGetString(prhs[0], trainFileName, mxGetN(prhs[0]) + 1);
68 
69  // input 2 - model file
70  mxGetString(prhs[1], modelFileName, mxGetN(prhs[1]) + 1);
71 
72  // input 3 - parameters
73  if (nrhs == 2)
74  parseInputMatlab(&param, NULL, true, trainFileName, modelFileName);
75  else
76  {
77  mxGetString(prhs[2], inputParamString, mxGetN(prhs[2]) + 1);
78  parseInputMatlab(&param, inputParamString, true, trainFileName, modelFileName);
79  }
80 
81  if (param.RANDOMIZE == 0)
82  srand(0);
83 
84  // train a model
85  switch (param.ALGORITHM)
86  {
87  case PEGASOS:
88  model = new budgetedModelAMM;
89  trainData = new budgetedData(trainFileName, param.DIMENSION - (int) (param.BIAS_TERM != 0.0), param.CHUNK_SIZE);
90  trainPegasos(trainData, &param, (budgetedModelAMM*) model);
91  break;
92  case AMM_BATCH:
93  model = new budgetedModelAMM;
94  trainData = new budgetedData(trainFileName, param.DIMENSION - (int) (param.BIAS_TERM != 0.0), param.CHUNK_SIZE, true);
95  trainAMMbatch(trainData, &param, (budgetedModelAMM*) model);
96  break;
97  case AMM_ONLINE:
98  model = new budgetedModelAMM;
99  trainData = new budgetedData(trainFileName, param.DIMENSION - (int) (param.BIAS_TERM != 0.0), param.CHUNK_SIZE);
100  trainAMMonline(trainData, &param, (budgetedModelAMM*) model);
101  break;
102  case LLSVM:
103  model = new budgetedModelLLSVM;
104  trainData = new budgetedData(trainFileName, param.DIMENSION - (int) (param.BIAS_TERM != 0.0), param.CHUNK_SIZE);
105  trainLLSVM(trainData, &param, (budgetedModelLLSVM*) model);
106  break;
107  case BSGD:
108  model = new budgetedModelBSGD;
109  trainData = new budgetedData(trainFileName, param.DIMENSION - (int) (param.BIAS_TERM != 0.0), param.CHUNK_SIZE);
110  trainBSGD(trainData, &param, (budgetedModelBSGD*) model);
111  break;
112  }
113 
114  // save model to .txt file
115  model->saveToTextFile(modelFileName, &(trainData->yLabels), &param);
116  delete model;
117 
118  // delete trainData, no more need for it
119  delete trainData;
120  }
121  else
122  {
123  budgetedModelMatlab *model;
124  budgetedDataMatlab *trainData = NULL;
125 
126  // GET ALL INPUTS
127  // inputs 1 and 2 - label vector and data points, loaded in the switch below
128 
129  // input 3 - parameters
130  if (nrhs == 2)
131  parseInputMatlab(&param, NULL, true);
132  else
133  {
134  mxGetString(prhs[2], inputParamString, mxGetN(prhs[2]) + 1);
135  parseInputMatlab(&param, inputParamString, true);
136  }
137 
138  // if randomization is switched off seed the RNG with a constant
139  if (param.RANDOMIZE == 0)
140  srand(0);
141 
142  // train a model
143  switch (param.ALGORITHM)
144  {
145  case PEGASOS:
146  model = new budgetedModelMatlabAMM;
147  trainData = new budgetedDataMatlab(prhs[0], prhs[1], &param);
148  trainPegasos(trainData, &param, (budgetedModelMatlabAMM*) model);
149  break;
150  case AMM_BATCH:
151  model = new budgetedModelMatlabAMM;
152  trainData = new budgetedDataMatlab(prhs[0], prhs[1], &param, true);
153  trainAMMbatch(trainData, &param, (budgetedModelMatlabAMM*) model);
154  break;
155  case AMM_ONLINE:
156  model = new budgetedModelMatlabAMM;
157  trainData = new budgetedDataMatlab(prhs[0], prhs[1], &param);
158  trainAMMonline(trainData, &param, (budgetedModelMatlabAMM*) model);
159  break;
160  case LLSVM:
161  model = new budgetedModelMatlabLLSVM;
162  trainData = new budgetedDataMatlab(prhs[0], prhs[1], &param);
163  trainLLSVM(trainData, &param, (budgetedModelMatlabLLSVM*) model);
164  break;
165  case BSGD:
166  model = new budgetedModelMatlabBSGD;
167  trainData = new budgetedDataMatlab(prhs[0], prhs[1], &param);
168  trainBSGD(trainData, &param, (budgetedModelMatlabBSGD*) model);
169  break;
170  }
171 
172  // save model to matlab structure
173  model->saveToMatlabStruct(plhs, &(trainData->yLabels), &param);
174  delete model;
175 
176  // delete trainData, no more need for it
177  delete trainData;
178  }
179 
180  // defragment the MATLAB space, not necessary but helps when working with large data,
181  // Matlab memory space gets very fragmented
182  mexCallMATLAB(0, 0, 0, 0, "pack");
183 }
