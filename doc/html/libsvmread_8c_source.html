42 static char* readline(FILE *input)
43 {
44  int len;
45 
46  if(fgets(line,max_line_len,input) == NULL)
47  return NULL;
48 
49  while(strrchr(line,'\n') == NULL)
50  {
51  max_line_len *= 2;
52  line = (char *) realloc(line, max_line_len);
53  len = (int) strlen(line);
54  if(fgets(line+len,max_line_len-len,input) == NULL)
55  break;
56  }

61 void read_problem(const char *filename, mxArray *plhs[])
62 {
63  int max_index, min_index, inst_max_index, i;
64  long elements, k;
65  FILE *fp = fopen(filename,"r");
66  int l = 0;
67  char *endptr;
68  mwIndex *ir, *jc;
69  double *labels, *samples;
70 
71  if(fp == NULL)
72  {
73  mexPrintf("can't open input file %s\n",filename);
74  fake_answer(plhs);
75  return;
76  }
77 
78  max_line_len = 1024;
79  line = (char *) malloc(max_line_len*sizeof(char));
80 
81  max_index = 0;
82  min_index = 1; // our index starts from 1
83  elements = 0;
84  while(readline(fp) != NULL)
85  {
86  char *idx, *val;
87  // features
88  int index = 0;
89 
90  inst_max_index = -1; // strtol gives 0 if wrong format, and precomputed kernel has <index> start from 0
91  strtok(line," \t"); // label
92  while (1)
93  {
94  idx = strtok(NULL,":"); // index:value
95  val = strtok(NULL," \t");
96  if(val == NULL)
97  break;
98 
99  errno = 0;
100  index = (int) strtol(idx,&endptr,10);
101  if(endptr == idx || errno != 0 || *endptr != '\0' || index <= inst_max_index)
102  {
103  mexPrintf("Wrong input format at line %d\n",l+1);
104  fake_answer(plhs);
105  return;
106  }
107  else
108  inst_max_index = index;
109 
110  min_index = min(min_index, index);
111  elements++;
112  }
113  max_index = max(max_index, inst_max_index);
114  l++;
115  }
116  rewind(fp);
117 
118  // y
119  plhs[0] = mxCreateDoubleMatrix(l, 1, mxREAL);
120  // x^T
121  if (min_index <= 0)
122  plhs[1] = mxCreateSparse(max_index-min_index+1, l, elements, mxREAL);
123  else
124  plhs[1] = mxCreateSparse(max_index, l, elements, mxREAL);
125 
126  labels = mxGetPr(plhs[0]);
127  samples = mxGetPr(plhs[1]);
128  ir = mxGetIr(plhs[1]);
129  jc = mxGetJc(plhs[1]);
130 
131  k=0;
132  for(i=0;i<l;i++)
133  {
134  char *idx, *val, *label;
135  jc[i] = k;
136 
137  readline(fp);
138 
139  label = strtok(line," \t\n");
140  if(label == NULL)
141  {
142  mexPrintf("Empty line at line %d\n",i+1);
143  fake_answer(plhs);
144  return;
145  }
146  labels[i] = strtod(label,&endptr);
147  if(endptr == label || *endptr != '\0')
148  {
149  mexPrintf("Wrong input format at line %d\n",i+1);
150  fake_answer(plhs);
151  return;
152  }
153 
154  // features
155  while(1)
156  {
157  idx = strtok(NULL,":");
158  val = strtok(NULL," \t");
159  if(val == NULL)
160  break;
161 
162  ir[k] = (mwIndex) (strtol(idx,&endptr,10) - min_index); // precomputed kernel has <index> start from 0
163 
164  errno = 0;
165  samples[k] = strtod(val,&endptr);
166  if (endptr == val || errno != 0 || (*endptr != '\0' && !isspace(*endptr)))
167  {
168  mexPrintf("Wrong input format at line %d\n",i+1);
169  fake_answer(plhs);
170  return;
171  }
172  ++k;
173  }
174  }
175  jc[l] = k;
176 
177  fclose(fp);
178  free(line);
179 
180  {
181  mxArray *rhs[1], *lhs[1];
182  rhs[0] = plhs[1];
183  if(mexCallMATLAB(1, lhs, 1, rhs, "transpose"))
184  {
185  mexPrintf("Error: cannot transpose problem\n");
186  fake_answer(plhs);
187  return;
188  }
189  plhs[1] = lhs[0];
190  }
191 }
